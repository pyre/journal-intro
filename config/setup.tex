\def\python#1#2{%
  \lstinputlisting[%
    language=python,
    morekeywords={as,self,yield,False,True,None},
    % pyre
    emph={[1]pyre,journal,component,protocol,application,properties},
    emphstyle={[1]\color{pyre@green}},
    %
    escapeinside={\#@}{@},
    #1]{#2}}

\def\Cpp#1#2{
  \lstinputlisting[%
    language=C++,
    % pyre
    emph={[1]pyre,journal},
    emphstyle={[1]\color{pyre@green}},
    %
    escapeinside={/*}{*/},
    #1]{#2}}
